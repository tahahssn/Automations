# Updating machine to latest versions
sudo apt update && sudo apt upgrade -y

# Install docker to machine
sudo apt  install -y docker.io  
# Check docker
docker --version

# Install docker compose to machine
sudo apt install -y docker-compose
# Check docker compose
docker compose --version


# Create a dedicated folder
mkdir ~/n8n && cd ~/n8n

# Create the compose file
nano docker-compose.yml

# Paste from docker-compose.yml file don't forget to update your public IP for that 

# Add ubuntu user to docker group
sudo usermod -aG docker ubuntu

# Apply the group change to current session
newgrp docker

# Verify the group is applied
groups

# Start n8n in the background
docker-compose up -d

# Check it's running
docker ps

# View live logs
docker-compose logs -f
